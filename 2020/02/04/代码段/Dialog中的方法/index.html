LoadEntity

loadEntity(entity: InventoryAdjustRow) {
            super.loadEntity(entity);
            var self = this;
            if (entity.InventoryAdjustTypeId == 1) {
                this.element.find(".field.StockLocationId").show();
            }
            if (!self.isNew()) {
                if (entity.InventoryStatusId == 0) {
                    this.element.find(".tool-button.confirm-button").removeClass("hide");
                } else {
                    if (!this.element.find(".tool-button.confirm-button").hasClass("hide")) {
                        this.element.find(".tool-button.confirm-button").addClass("hide");
                    }
                    Serenity.EditorUtils.setReadonly(this.element.find('.editor'), true);

                    self.saveAndCloseButton.hide();
                    self.applyChangesButton.hide();
                }               
                if (entity.InventoryStatusId == 3) {
                    this.element.find(".s-DialogToolbar").addClass("hide");                 
                }
            }
            Serenity.TabsExtensions.setDisabled(this.tabs, 'Orders', this.entity.InventoryStatusId == 0);
            if (this.entity.InventoryStatusId > 0) {
                Serenity.TabsExtensions.selectTab(this.tabs, 'Orders');
            }
            this.adjustLineGrid.inventoryEntity = entity;
            this.adjustLineGrid.myCallback = () => {
                self.reloadById();
            }
        }

加入Button

protected getToolbarButtons(): Serenity.ToolButton[] {

            let buttons = super.getToolbarButtons();
            var self = this;
            buttons.push({
                title: Q.text("Db.Stock.InventoryAdjust.StartInventory"),
                cssClass: 'confirm-button hide',              
                onClick: () => {
                    if (self.entity.InventoryAdjustTypeId == 1)//AllProductOfOneLocation
                    {
                        self.creatInventoryLineByLocationId(this.entityId, this.entity.StockLocationId);
                    } else {
                        self.setInventoryAdjustStatus(this.entityId, 1);
                    }

                }
            });         
            return buttons;
        }

Template Dialog

namespace Mom.Production {
    export class MaterialPrepareDetailDialog extends Serenity.TemplatedDialog<any> {

        public prodOrderItemBomIds: string[];
        private materialPrepareDetailGrid: Plan.MaterialPrepareDetailGrid;
        constructor() {
            super();
            var self = this;
            this.set_dialogTitle("生产订单释放检查");
            this.materialPrepareDetailGrid = new Plan.MaterialPrepareDetailGrid(this.byId("TemplateGrid"));
        }

        dialogOpen() {
            super.dialogOpen();
            this.materialPrepareDetailGrid.prodOrderItemBomIds = this.prodOrderItemBomIds;
            this.materialPrepareDetailGrid.init();
            Q.initFullHeightGridPage(this.byId("TemplateGrid"));
        }

        protected getDialogOptions(): JQueryUI.DialogOptions {
            let opt = super.getDialogOptions();
            opt.width = 1200;
            opt.modal = false;
            return opt;
        }

        protected getTemplate() {
            return `<div  class="s-DialogContent" style="padding: 10px 15px">
                        <div class="row">
                            <div class="col-md-12">
                                <div id="~_TemplateGrid" style="min-height:500px"></div>
                            </div>
                        </div>

                    </div>`;
        }
    }
